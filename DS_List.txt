#include <stdio.h>
int findMissingNumber(int arr[], int size)
{
  int totalSum = (size + 1) * (size + 2) / 2;
  int i, currentSum = 0;

  for (i = 0; i < size; i++)
  {
    currentSum += arr[i];
  }

  return totalSum - currentSum;
}
